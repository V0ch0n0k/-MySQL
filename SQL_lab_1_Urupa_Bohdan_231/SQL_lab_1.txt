DROP DATABASE IF EXISTS LibraryBD;

CREATE TABLE BookTable(
    num_ INT NOT NULL PRIMARY KEY,
    code_ INT NOT NULL,
    novelty_ VARCHAR(3) NOT NULL CHECK (novelty_ = 'Yes' OR novelty_ = 'No'),
    name_ VARCHAR(100) NOT NULL,
    price_ FLOAT(6, 2),
    publisher_ VARCHAR(100) NOT NULL,
    pages_ INT NOT NULL CHECK (pages_ > 0),
    formar_ VARCHAR(30) NOT NULL,
    date_ DATE,
    circulation_ INT NOT NULL DEFAULT 5000 CHECK (circulation_ > 0) ,
    subject_ VARCHAR(30) NOT NULL,
    category_ VARCHAR(30) NOT NULL
);

CREATE INDEX idx_Book ON BookTable(num_ ASC ,name_ ASC, subject_ ASC, category_ ASC);
